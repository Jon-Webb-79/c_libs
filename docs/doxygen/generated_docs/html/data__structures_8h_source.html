23/*
24 * This macro will push a scalar data point to a user defined index within an
25 * existing vector. The macro only selects the correct function, and therefore
26 * does not incease memory usage or decrease code execution time. While the user
27 * can push data to any index, the macro will execute fastest when the user
28 * pushes data to the next available index.
29 *
30 * /param vec A vector struct container
31 * /param value The data point to be pushed to the vector
32 * /param index The index where data is to be inserted
33 *
34 * @code
35 * Int vec;
36 * INIT_VECTOR(vec, 20);
37 * PUSH(vec, 1, vec.active_length);
38 * PUSH(vec, 2, vec.active_length);
39 * PUSH(vec, 3, vec.active_length);
40 * PUSH(vec, 10, 0);
41 * for (size_t i = 0; i < vec.active_length; i++) {
42 * printf("%d\n", vec.array[i]);
43 * }
44 * // >> 10 1 2 3
45 * @endcode
46 */
47#define PUSH(T, value, index) _Generic( (T), Short: push_short_vector, \
48 UShort: push_ushort_vector, \
49 Int: push_int_vector, \
50 UInt: push_uint_vector, \
51 LInt: push_long_vector, \
52 ULInt: push_ulong_vector, \
